Object Concepts in JavaScript

Encapsulation Example

// Encapsulation Example:
let student = {
    name: "Ahmed",
    age: 12,
    level: 5,
    hello: function() {
        return `Hello, I am ${this.name}, ${this.age} years old.`;
    },
    pass: function() {
        this.age++;
        this.level++;
    }
};
student.pass();
console.log(student.hello());

Constructor Function Example

// Constructor Example:
function Student(name, age, level) {
    this.name = name;
    this.age = age;
    this.level = level;
    this.hello = function() {
        return `Hello, I am ${this.name}, ${this.age} years old.`;
    };
    this.pass = function() {
        this.age++;
        this.level++;
    };
}
let student = new Student("Ahmed", 12, 5);
student.pass();
console.log(student.hello());

Date Object Example

// Date Object Example:
let currentDate = new Date();
let specificDate = new Date(1999, 2, 14, 4, 20, 30);
let dob = new Date('1999-05-12 04:20:30');
console.log(currentDate);
console.log(specificDate);
console.log(dob);
console.log(dob.getFullYear());
console.log(dob.getMonth() + 1);
console.log(dob.getDate());

Regular Expressions Example

// Regular Expressions Example:
let regex = /hello/i;
let str = "Hello World!";
console.log(regex.test(str)); // true
console.log(str.match(regex)); // ["Hello"]
